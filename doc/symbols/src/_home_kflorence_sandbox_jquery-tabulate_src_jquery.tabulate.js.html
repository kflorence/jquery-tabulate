 44 /**
 45  * Anonymous function wrapper.
 46  * 
 47  * <p>
 48  *   Creates closure for the jQuery.tabulate plugin.
 49  * </p>
 50  * 
 51  * @exports $ as jQuery
 52  * @param $ The jQuery object

 54 ;(function($) {    
 55     /**
 56      * The jQuery.tabulate plugin.
 57      * 
 58      * <p>
 59      *   Tabulate is a jQuery plugin that takes arbitrary data, divides it into
 60      *   rows and columns, then builds a table around it.  It is meant to be
 61      *   very flexible and extensible and is purposely vague in some areas to
 62      *   accommodate these goals.
 63      * </p>
 64      * 
 65      * @example $('#element').tabulate({...});
 66      * @param {object} options An object containing default option overrides.
 67      * @return {jQuery} The jQuery object that was passed to this function.
 68      */
 69     $.fn.tabulate = function(options) {
 70         return this.each(function() {
 71             var $this = $(this);
 72 
 73             // don't re-create if this element has already been tabulated
 74             if (!$this.data("tabulate")) {
 75                 /**
 76                  * The tabulate class instance.
 77                  * 
 78                  * @see jQuery.tabulate
 79                  * @name tabulate
 80                  * @type Object
 81                  * @class
 82                  */
 83                 var tabulate = $.extend(true, {}, $.tabulate);
 84 
 85                 // store class instance in $this
 86                 $this.data("tabulate", tabulate);
 87 
 88                 // initialize, pass in options
 89                 tabulate.init($this, options);
 90             }
 91         });
 92     };

719         /**
720          * Initializes the tabulate class.
721          * 
722          * <p>
723          *   Merges default options with those passed in, builds the table
724          *   skeleton, creates jQuery objects from {@link jQuery.tabulate.options.elements}
725          *   and binds any needed event handlers.  Finally, triggers
726          *   {@link jQuery.tabulate.event_handlers.event:post_init}.
727          * </p>
728          *
729          * @param jQuery $wrapper The element(s) passed in from {@link jQuery.fn.tabulate}.
730          * @param object options Options to overwrite the default ones with.
731          */
732         init: function($wrapper, options) {            
733             var self = this;
734 
735             // extend default options, then attach to this instance
736             $.extend(true, this.options, options || {});
737             $.extend(true, this, this.options);
738             $.extend(this, {
739                 $navigation: this.fragments.$navigation.clone(),
740                 $container: this.fragments.$container.clone(),
741                 $wrapper: $wrapper
742             });
743             
744             // build container
745             this.$wrapper.append(this.$container).append(this.$navigation);
746             
747             // generate full themes path
748             this.paths.theme = [this.paths.tabulate, this.paths.theme].join("/");
749 
750             // set limit to first item in results per page array, if not set
751             this.filters.limit = this.filters.limit || this.results_per_page[0];
752             
753             // Build out elements
754             $.each(this.elements, function(key, value) {
755                 self.elements[key] = $(value, $wrapper);
756             });
757 
758             // navigation: previous
759             if (this.elements.$previous.length) {
760                 this.elements.$previous.attr({
761                     src: [this.paths.theme, this.paths.images.previous].join("/"),
762                     alt: "Previous Page",
763                     title: "Previous Page"
764                 }).click(function() {
765                     self.previous(this); return false;
766                 });
767             }
768 
769             // navigation: next
770             if (this.elements.$next.length) {
771                 this.elements.$next.attr({
772                     src: [this.paths.theme, this.paths.images.next].join("/"),
773                     alt: "Next Page",
774                     title: "Next Page"
775                 }).click(function() {
776                     self.next(this); return false;
777                 });
778             }
779 
780             // navigation: current page
781             if (this.elements.$current_page.length) {
782                 this.elements.$current_page.keyup(function(event) {
