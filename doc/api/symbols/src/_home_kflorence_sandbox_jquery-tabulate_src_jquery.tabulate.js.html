621         /**
622          * Initializes the tabulate class.
623          * 
624          * <p>
625          *   Merges default options with those passed in, builds the table
626          *   skeleton, creates jQuery objects from {@link jQuery.tabulate.options.elements}
627          *   and binds any needed event handlers.  Finally, triggers
628          *   {@link jQuery.tabulate.event_handlers.event:post_init}.
629          * </p>
630          *
631          * @param jQuery $wrapper The element(s) passed in from {@link jQuery.fn.tabulate}.
632          * @param object options Options to overwrite the default ones with.
633          */
634         init: function($wrapper, options) {            
635             var self = this;
636 
637             // extend default options, then attach to this instance
638             $.extend(true, this.options, options || {});
639             $.extend(true, this, this.options);
640 
641             this.$wrapper = $wrapper;
642 
643             // store class instance in wrapper
644             this.$wrapper.data("tabulate", this);
645 
646             // append theme path to tabulate path (no trailing "/")
647             this.paths.theme = [this.paths.tabulate, this.paths.theme].join("/");
648 
649             // set limit to first item in results per page array, if not set
650             this.data.filters.limit = this.data.filters.limit || this.results_per_page[0];
651 
652             // create table
653             this.$wrapper.append(this.$table = this.$fragments.table)
654                 .append(this.$navigation = this.$fragments.navigation);
655 
656             // build table sections
657             $.each(this.table.columns, function(section, columns) {
658                 self.$table[section] = self.$fragments[section]
659                 self.$table.append(self.$table[section]);
660             });
661 
662             // build jQuery objects from selectors
663             $.each(this.$elements, function(key, selector) {
664                 self.$elements[key] = $(selector);
665             });
666 
667             // previous
668             if (this.$elements.previous.length) {
669                 this.$elements.previous.attr({
670                     src: this.paths.theme + "/prev.gif",
671                     alt: "Previous Page",
672                     title: "Previous Page"
673                 }).click(function() {
674                     self.previous(this); return false;
675                 });
676             }
677 
678             // next
679             if (this.$elements.next.length) {
680                 this.$elements.next.attr({
681                     src: this.paths.theme + "/next.gif",
682                     alt: "Next Page",
683                     title: "Next Page"
684                 }).click(function() {
685                     self.next(this); return false;
686                 });
687             }
688 
689             // current page
690             if (this.$elements.current_page.length) {
691                 this.$elements.current_page.keyup(function(event) {
692                     var page = parseInt($(this).val());
693 
694                     if (!isNaN(page) && event.which == 13) {
695                         self.go_to(this, page);
696                     }
697                 });
698             }
699 
700             // results per page
701             if (this.$elements.results_per_page.length) {
702                 $.each(this.results_per_page, function(i, value) {
703                     self.$elements.results_per_page.append(
704                         self.$fragments.option.clone().val(value).text(value)
705                     );
706                 });
707 
708                 this.$elements.results_per_page.change(function() {
709                     var limit = parseInt($(this).val());
710 
711                     if (!isNaN(limit)) {
712                         self.update_filters({limit: limit});
713                     }
714                 });
715             }
716 
717             // bind event handlers
718             $.each(this.event_handlers, function(name, handler) {
719                 self.$wrapper.bind([name, self.name].join("."), function() {
720                     handler.apply(self, arguments);
721                 });
722             });
723 
724             // fire init handler
725             this.trigger("post_init");
726         },

728         /**

734          *   go to, then decrements current_page. Finally, it triggers
735          *   {@link jQuery.tabulate.event_handlers.event:refresh}.
736          * </p>

738          * @param element The DOM element that was clicked.

743

771         /**

773          * page.
774          * 
775          * <p>
776          *   This function first checks to make sure the specified page exists,
777          *   then sets current_page to its value. Finally, it triggers
778          *   {@link jQuery.tabulate.event_handlers.event:refresh}.

780          * 
781          * @param element element The DOM element that was clicked.

784         go_to: function(element, page) {
785             if (page >= 1 && page <= this.total_pages) {                
786                 this.current_page = page;

859          *   {@link jQuery.tabulate.event_handlers.event:loading} with <em>true</em>
860          *   as its argument, effectively turning it on. After the data type is
861          *   determined and the data has been gathered this function will
862          *   trigger {@link jQuery.tabulate.event_handlers.event:post_load} with the
863          *   data object as its argument.  It is important to note that if both
864          *   an AJAX request and a JSON object are passed to this function, the
865          *   result of the two will be merged together before being passed to
866          *   the post_load event handler.
867          * </p>
868          * 
869          * @see jQuery.tabulate.options.data
870          * @see jQuery.tabulate.event_handlers
871          * 
872          * @param Object request The request object.
873          * @param Object filters Filters that will apply to this request only.
874          */

880             this.trigger("loading", true);
881 
882             if (request.ajax) {
883                 filters = $.extend(true, {}, this.data.filters, filters);
884 
885                 // merge request with additional internal arguments
886                 var request = $.extend(true, {}, request.ajax, {
887                     data: filters,

890                         $.extend(true, data, request.json || {});

894                     },
895                     error: function() {

897                     }
898                 });

901             }

905                     data = request.json;
906                 }
907 
908                 this.trigger("post_load", data);

912         /**
913          * Applies properties to rows, columns and cells in the table.

915          * <p>
916          *   Each row or column in the table may have any number of properties
917          *   assigned to it.  These properties are passed into the plugin upon
918          *   initialization as {@link jQuery.tabulate.options.rows} for rows, and
919          *   {@link jQuery.tabulate.options.columns} for columns.  The following
920          *   outlines valid properties for both rows and columns (note that
921          *   "cell" refers to a column within a row):
922          * </p>

924          * <ul>

927          *     or row.  In the case of an object, this would be the key.  In the
928          *     case of an Array, this would be the index.
929          *   </li>
930          *   <li>

934          *   </li>
935          *   <li>
936          *     <strong>args</strong> - An array of arguments to pass to the
937          *     callback function via javaScript's apply function.
938          *   </li>
939          *   <li>
940          *     <p>
941          *       <strong>callback</strong> - This is a function that will be
942          *       called when the cell or row is instantiated.  This is useful
943          *       for manipulating something based on the data assigned to a cell
944          *       or row.  Inside of the this function, <em>this</em> is equal to
945          *       {@link jQuery.tabulate}.  The arguments passed to this function vary
946          *       between cells and rows.
947          *     </p>
948          *     
949          *     <p>The arguments passed for a row are:</p>
950          * 
951          *     <ul>
952          *       <li>
953          *         <strong>$row</strong> - The jQuery object representing the
954          *         current row.
955          *       </li>
956          *       <li>
957          *         <strong>row_data</strong> - The data associated with the current
958          *         row.
959          *       </li>
960          *     </ul>
961          * 
962          *     <p>The arguments passed for a cell are:</p>
963          * 
964          *     <ul>
965          *       <li>
966          *         <strong>$row</strong> - The jQuery object representing the
967          *         current row.
968          *       </li>
969          *       <li>
970          *         <strong>$cell</strong> - The jQuery object representing the
971          *         current cell.
972          *       </li>
973          *       <li>
974          *         <strong>$content</strong> - The jQuery object representing
975          *         $cell's content wrapper.
976          *       </li>
977          *       <li>
978          *         <strong>row_data</strong> - The data associated with the
979          *         current row.
980          *       </li>
981          *     </ul>
982          *   </li>
983          *   <li>
984          *     <strong>event_handlers</strong> - An object containing event
985          *     handlers to bind to the cell or row.  These should be defined in
986          *     the same format as {@link jQuery.tabulate.event_handlers}.
987          *   </li>
988          *   <li>
989          *     <strong>styles</strong> - An object containing style information
990          *     to be applied to the row or cell. This object is passed to the
991          *     <a href="http://api.jquery.com/jQuery.css/">jQuery.css</a>
992          *     function.
993          *   </li>
994          * </ul>
995          * 
996          * <p>
997          *   Columns may also contain these properties, in addition to those
998          *   defined above:
999          * </p>
1000          * 
1001          * <ul>
1002          *   <li>
1003          *     <strong>$content</strong> - The jQuery object in which to append
1004          *     content inside of. If not given, defaults to $element.
1005          *   </li>
1006          *   <li>
1007          *     <strong>content</strong> - The content to append inside $content.
1008          *     Content can be passed in as a string, a jQuery object, or as a
1009          *     key within the dataset <em>data</em>. If you wish to pass in a
1010          *     key, you may do so within curly braces like so: "{some_key}".  If
1011          *     the key is nested within the dataset, you may separate each level
1012          *     by use of a dot like so: "{obj.prop.key}".
1013          *   </li>
1014          * </ul>
1015          * 
1016          * @see <a href="http://api.jquery.com/jQuery.css/">jQuery.css</a>
1017          * @see <a href="https://developer.mozilla.org/en/Core_JavaScript_1.5_Reference/Global_Objects/Function/apply">Function.apply</a>
1018          * @see jQuery.tabulate.options.rows
1019          * @see jQuery.tabulate.options.columns
1020          * @see jQuery.tabulate.event_handlers
1021          * 
1022          * @param jQuery $element The element to apply the properties to.
1023          * @param Object properties An object containing the properties.
1024          * 
1025          * @return jQuery The modified element
1026          */
1027         apply_properties: function($element, settings, properties) {
1028             var self = this,
1029                 properties = properties || {},
1030                 dataset = settings.dataset || {},
1031                 key = (typeof settings.key != "undefined" ? settings.key : "");
1032 
1033             $element.each(function(i, element) {
1034                 var $element = $(element),
1035                     data = (dataset[key] ? dataset[key] : dataset),
1036                     name = ($.isNumber(key) ? parseInt(key) + 1 : key),
1037                     $content = $(".tabulate-content", $element),
1038                     args = [$element, $content, data];
1039 
1040                 // content defaults to element if not present
1041                 if (!$content.length) $content = $element;
1042 
1043                 // apply type class
1044                 if (typeof settings.type == "string") {
1045                     $element.addClass([self.name, settings.type, name].join("-"));
1046                 }
1047 
1048                 // set up properties according to type
1049                 switch(typeof properties) {
1050                     case "string": {
1051                         properties = { content: properties };
1052                         break;
1053                     }  
1054                     case "function": {
1055                         properties = { content: properties.apply(self, args) };
1056                         break;
1057                     }
1058                 }
1059 
1060                 // properties must be an object
1061                 if (typeof properties == "object") {
1062                     $.each(properties, function(property, value) {
1063                         switch(property) {
1064                             case "attributes": {
1065                                 $element.attr(value);
1066                                 break;
1067                             }
1068                             case "content": {
1069                                 if (typeof value == "function") {
1070                                     value = value.apply(self, args);
1071                                 }
1072 
1073                                 switch(typeof value) {
1074                                     case "object": {
1075                                         if (value instanceof $) {
1076                                             $content.append(value);
1077                                         }
1078                                         break;
1079                                     }
1080                                     default: {
1081                                         $content.append(value.replace(self.parse_key(), function(str, key) {
1082                                             return $.getObject(key, dataset) || "";
1083                                         }));
1084                                         break;
1085                                     }
1086                                 }
1087                                 break;
1088                             }
1089                             case "class_name": {
1090                                 switch (typeof value) {
1091                                     case "function": {
1092                                         $element.addClass(value.apply(self, args));
1093                                         break;
1094                                     }
1095                                     default: {
1096                                         $element.addClass(value);
1097                                         break;
1098                                     }
1099                                 }
1100                                 break;
1101                             }
1102                             case "events": {
1103                                 $.each(value, function(name, handler) {
1104                                     $element.bind(name, function(event) {
1105                                         handler.apply(self, [event].concat(args));
1106                                     });
1107                                 });
1108                                 break;
1109                             }
1110                             case "filter": {
1111                                 value.apply(self, args);
1112                                 break;
1113                             }
1114                             case "styles": {
1115                                 $element.css(value);
1116                                 break;
1117                             }
1118                         }
1119                     });
1120                 }
1121             });
